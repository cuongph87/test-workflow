flowchart TD
    start(["User taps Deposit
(Beans App)"]) --> buildTx

    buildTx["App builds Stellar transaction:
Payment / PathPaymentStrictSend /
PathPaymentStrictReceive"]
    buildTx --> signTx["App signs transaction
(client-side, user's key)"]
    signTx --> apiCall

    apiCall["POST /v4/me/cards/{cardId}/deposits
Header: Idempotency-Key
Body: amount + signedTransactionXdr"]

    apiCall --> idempotency{"Existing deposit
for idempotency key?"}
    idempotency -- "Yes" --> returnExisting(["Return existing
card transaction"])
    idempotency -- "No" --> validate

    validate{"Validate:
card ownership,
amount > 0,
XDR parseable,
signature threshold"}
    validate -- "Fails" --> errorResponse(["400 / 404
Error response"])
    validate -- "Passes" --> atomicWrite

    atomicWrite["Atomic DB Write:
1. card_deposits (Draft)
2. card_transactions (Draft)
3. Outbox job (2 steps)"]
    atomicWrite --> apiResponse(["200 OK
Returns card transaction"])
    atomicWrite --> step0

    %% ── OUTBOX STEP 0 ──────────────────────────────────────
    step0["Outbox Step 0:
SubmitTransactionToStellarStepRunHandler"]
    step0 --> feeBump["horizonService.FeeBumpTransaction:
Wrap in fee bump envelope
(Beans sponsors fees)
Sign + submit to Stellar"]
    feeBump --> step0Result{"Stellar submission
result?"}

    step0Result -- "Success
(tx hash returned)" --> step1
    step0Result -- "FeeBumpInnerTransaction
FailedException" --> immediateFailure["Immediate failure
(no retries, deterministic)"]
    step0Result -- "Other exception
(network timeout)" --> retry{"Retry count
< 3?"}
    retry -- "Yes" --> step0
    retry -- "No (exhausted)" --> failedHandler

    immediateFailure --> failedHandler

    %% ── OUTBOX STEP 1 ──────────────────────────────────────
    step1["Outbox Step 1:
SyncDatabaseCardDepositRecordsStepRunHandler"]
    step1 --> syncDb["Update deposit:
StellarTransactionHash + Status → Pending
Update card_transaction:
Status → Pending"]
    syncDb --> awaitWebhook(["Deposit now Pending
Awaiting Kulipa webhook"])

    %% ── PERMANENT FAILURE ──────────────────────────────────
    failedHandler["CreateCardDepositJobRunFailedHandler"]
    failedHandler --> markFailed["Deposit status → Failed
Transaction status → Failed"]
    markFailed --> notifyFailed["Push notification:
deposit failed"]
    notifyFailed --> terminalFailed(["Terminal: Failed"])

    %% ── WEBHOOK: CONFIRMED ─────────────────────────────────
    awaitWebhook --> webhookConfirmed

    webhookConfirmed["Webhook received:
blockchainTransaction.confirmed
(type: Deposit)"]
    webhookConfirmed --> parseConfirmed["BlockchainTransactionConfirmedEventProcessor
Parse → BlockchainTransactionConfirmedEventData"]
    parseConfirmed --> lookupUser["Find AnchorUser by ExternalId
Fetch + update card balance"]
    lookupUser --> matchDeposit{"Match deposit by
StellarTransactionHash?"}

    matchDeposit -- "Matched
(in-app deposit)" --> guardConfirm{"Current
deposit status?"}
    matchDeposit -- "Not matched
(external deposit)" --> externalDeposit

    guardConfirm -- "Confirmed" --> skipConfirm(["Skip:
already confirmed"])
    guardConfirm -- "Failed" --> skipTerminal(["Skip + log warning:
do not transition Failed back"])
    guardConfirm -- "Draft / Pending" --> confirmUpdate["Update deposit:
ExternalId + ConfirmedAmount
Status → Confirmed
Update card_transaction:
Amount → confirmed amount
Status → Confirmed"]
    confirmUpdate --> notifyConfirmed["Push notification:
deposit confirmed"]
    notifyConfirmed --> terminalConfirmed(["Terminal: Confirmed"])

    %% ── EXTERNAL DEPOSIT ────────────────────────────────────
    externalDeposit["Create records on-the-fly:
card_deposits (Confirmed)
Amount = ConfirmedAmount = on-chain amount
IdempotencyKey = tx hash
card_transactions (Confirmed)"]
    externalDeposit --> notifyExternal["Push notification:
deposit confirmed"]
    notifyExternal --> terminalExternal(["Terminal: Confirmed
(external deposit)"])
